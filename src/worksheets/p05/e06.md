# Ejercicio 6

```
zip :: [a] -> [b]-> [(a, b)]
zip [] ys = []
zip (x : xs) [] = []
zip (x : xs) (y : ys) = (x, y) : zip xs ys

unzip :: [(a, b)] -> ([a], [b])
unzip [] = ([], [])
unzip ((x , y) : ps) = (x : xs, y : ys)
  where (xs, ys) = unzip ps

-- EXTRA

uncurry :: (a -> b -> c) -> ((a, b) -> c)
uncurry f p =  f (fst p) (snd p)

(.) :: (b -> c) -> (a -> b) -> a -> c
(.) f g = \x -> f (g x)
```
